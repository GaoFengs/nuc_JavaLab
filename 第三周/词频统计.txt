import java.util.*;
 
public class Frequency {
    public int getFrequency(String[] article, int n, String word) {
        // write code here
        int count = 0 ;
         
        for(int i = 0 , len = article.length ; i < len ; i ++  ){
            if(article[i].equals(word)){
                count ++;
            }
             
        }
         
        return count;
    }
}
